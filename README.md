\left| \text{QR} (p) \right| &=& 1 \ + \ \frac{p - 1}{2}\\

\text{if} \ e \notin	 2\mathbb{Z_+}, \ \left| \text{QR} (p^e) \right| &=& 1 \ + \ p\left| \text{QR}(p^{e - 1}) \right| \ + \ \lfloor \sqrt{p - 1} \rfloor	 \ + \ \left| \text{QR}_{\text{non-square}}(p) \right|
\end{eqnarray}$$
